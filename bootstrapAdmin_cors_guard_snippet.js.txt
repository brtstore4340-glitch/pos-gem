/*
   Recommended guard + CORS wrapper (add to bootstrapAdmin)
  - Handles OPTIONS
  - Enforces POST
  - Validates JSON body
  - Returns readable 400/405 instead of 500
  - Restricts CORS to whitelisted origins (rejects 403 if not allowed)
*/
const functions = require("firebase-functions");
const cors = require("cors");

// Whitelist of allowed origins
const allowedOrigins = [
  "http://localhost:5173",      // Local dev
  "http://127.0.0.1:5173",      // Local dev (IP)
  "https://boots-pos.example.com" // Production (update to your domain)
];

// CORS middleware with origin validation
const corsWithValidation = cors({
  origin: (origin, callback) => {
    // Allow requests with no origin (like mobile apps or curl requests)
    if (!origin) return callback(null, true);
    
    if (allowedOrigins.includes(origin)) {
      return callback(null, true);
    }
    
    // Reject non-whitelisted origins
    return callback(new Error(`CORS not allowed for origin: ${origin}`), false);
  },
  credentials: true,
  methods: ["GET", "POST", "OPTIONS"]
});

exports.bootstrapAdmin = functions.region("asia-southeast1").https.onRequest((req, res) => {
  corsWithValidation(req, res, async () => {
    try {
      if (req.method === "OPTIONS") return res.status(204).send("");
      if (req.method !== "POST") return res.status(405).json({ error: "Method Not Allowed" });

      const { email, password, username } = req.body || {};
      if (!email || !password || !username) {
        return res.status(400).json({ error: "Missing fields", need: ["email","password","username"] });
      }

      // ... your existing logic here ...

    } catch (e) {
      console.error("bootstrapAdmin error:", e);
      return res.status(500).json({ error: e?.message || String(e) });
    }
  });
});
